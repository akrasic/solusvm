def default_option(key)
  @@yaml ||= begin
    file = File.join(File.expand_path(ENV['HOME']), '.solusvm.yml')
    if File.exists?(file)
      YAML::load(File.open(file))
    else
      {}
    end
  end

  @@yaml[key.to_s]
end

def namespace(name=nil)
  if name
    super
  else
    @namespace ||= super.sub(/_cli$/, '')
  end
end
