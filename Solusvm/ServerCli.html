def switch(vserverid, switch_value, on_method, off_method)
  if switch_value == "on"
    api.send(on_method, vserverid)
  elsif switch_value == "off"
    api.send(off_method, vserverid)
  else
    "Invalid switch value. Valid values are on|off"
  end
end
